let story = 'Last weekend, I took literally the most beautiful bike ride of my life. The route is called "The 9W to Nyack" and it actually stretches all the way from Riverside Park in Manhattan to South Nyack, New Jersey. It\'s really an adventure from beginning to end! It is a 48 mile loop and it basically took me an entire day. I stopped at Riverbank State Park to take some extremely artsy photos. It was a short stop, though, because I had a really long way left to go. After a quick photo op at the very popular Little Red Lighthouse, I began my trek across the George Washington Bridge into New Jersey.  The GW is actually very long - 4,760 feet! I was already very tired by the time I got to the other side.  An hour later, I reached Greenbrook Nature Sanctuary, an extremely beautiful park along the coast of the Hudson.  Something that was very surprising to me was that near the end of the route you actually cross back into New York! At this point, you are very close to the end.';

let overusedWords = ['really', 'very', 'basically'];

let unnecessaryWords = ['extremely', 'literally', 'actually' ];



// ------------------------------TRANSFORMANDO A STRING EM ARRAY---------------------------


const storyArray = story.split(" ")
console.log(`You had ${storyArray.length} words`)


//---------------------------DEIXA AS PALAVRAS QUE NÃO SAO DESNECESSARIAS------------------
//--------------------------------------------------compara com unecessaryWords------------


const betterWords = storyArray.filter(char =>
{
//----------------------------------------------------se a palavra não está na array de--- 
//                                                    unnecesaryWords---------------------

  if (!unnecessaryWords.includes(char)) {
    return char;
  }
})

console.log(`You have ${betterWords.length} words now`);



//-----------------CONTANDO ALGUMAS PALAVRAS QUE USEI DEMAIS (OVERUSEDWORDS)---------------

//-----------------------------------------crio um objeto para salvar os valores repetidos- 



const repetition ={};
for (const word of overusedWords){
  repetition[word] = 0;
}


//---------------------------------------------------conto as palavras----------------------

betterWords.filter(char => {
  if (overusedWords.includes(char)) {
    return repetition[char]++
  }
})



//---------------------OBJECT TO STRING -----------------------------------------------------
//-------------------------------------JSON.stringify (obj) transforma objeto em string e---- 
//                                     me permite logá-lo em uma frase-----------------------

console.log (`You have ${JSON.stringify(repetition)} repeated words`)


//------------------------------------------HOW MANY SENTENCES-------------------------------
// ----------------------------------------calculo numero de sentenças-----------------------

let count = 0;
betterWords.forEach(char => {
  if (["!", "?", ";", "."].includes(char.slice(-1))) {
    count++;
  }
})

console.log(`you have ${count} sentences`);




//---------------------------- BACK TO STRING-------------------------------------------


//   console.log(betterWords.join(" "));


//+++++++++++++++++++++++++++++++  CHANGE WORDS TO STANDARD ++++++++++++++++++++++++++++
//++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++

//--------------------------------------tira char especificos como: . ! ? --------------
//--------------------------------------explanation in the comments!--------------------

standardWord = [];
for (const words of betterWords) {
  const a = words.toLowerCase().replace(/!|,|:|'|"/g, "").replace(".", "").replace("?", "")
  standardWord.push(a);
}
//console.log(standardWord);

//+++++++++++++++++++++++++  WORD THAT APPEARS THE MOST  ++++++++++++++++++++++++++++++++


mostWords ={};
for (const word of standardWord){
  mostWords[word] = (mostWords[word] || 0) + 1;
}

//console.log(mostWords);


//------------------------------------------- Object.keys (vê as propriedades------------- 
//                                            do objeto). Crio uma especie de array-------

var arrayMostWords = Object.keys(mostWords).map( propriedade => { 
  return mostWords[propriedade];
})


var min = Math.min.apply( null, arrayMostWords );
var max = Math.max.apply( null, arrayMostWords );

console.log( 'Min value: ' + min + ', max value: ' + max )
